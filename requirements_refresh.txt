# 报名工具 Token 自动刷新脚本依赖

# 核心依赖（已包含在主项目中）
requests>=2.28.0
pymongo>=4.0.0
mongoengine>=0.24.0

# 可选依赖（仅定时器模式需要）
schedule>=1.1.0

# 说明：
# - requests: HTTP 请求库
# - pymongo: MongoDB 驱动
# - mongoengine: MongoDB ORM
# - schedule: 定时任务库（可选，仅 --scheduler 模式需要）

# 安装方式：
# pip install -r requirements_refresh.txt

# 或只安装核心依赖：
# pip install requests pymongo mongoengine

# 如果需要内置定时器：
# pip install schedule
